698 template<typename SrcType>
699 void ofPixels_<PixelType>::copyFrom(const ofPixels_<SrcType> & mom){
700  if(mom.isAllocated()){
701  allocate(mom.getWidth(),mom.getHeight(),mom.getNumChannels());
702 
703  const float srcMax = ( (sizeof(SrcType) == sizeof(float) ) ? 1.f : std::numeric_limits<SrcType>::max() );
704  const float dstMax = ( (sizeof(PixelType) == sizeof(float) ) ? 1.f : std::numeric_limits<PixelType>::max() );
705  const float factor = dstMax / srcMax;
706 
707  if(sizeof(SrcType) == sizeof(float)) {
708  // coming from float we need a special case to clamp the values
709  for(size_t i = 0; i < mom.size(); i++){
710  pixels[i] = ofClamp(mom[i], 0, 1) * factor;
711  }
712  } else{
713  // everything else is a straight scaling
714  for(size_t i = 0; i < mom.size(); i++){
715  pixels[i] = mom[i] * factor;
716  }
717  }
718  }
719 }
