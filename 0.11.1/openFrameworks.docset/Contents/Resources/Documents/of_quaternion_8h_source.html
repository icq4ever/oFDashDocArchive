485 ofVec3f ofQuaternion::operator*(const ofVec3f& v) const {
486  // nVidia SDK implementation
487  ofVec3f uv, uuv;
488  ofVec3f qvec(_v.x, _v.y, _v.z);
489  uv = qvec.getCrossed(v); //uv = qvec ^ v;
490  uuv = qvec.getCrossed(uv); //uuv = qvec ^ uv;
491  uv *= (2.0f * _v.w);
492  uuv *= 2.0f;
493  return v + uv + uuv;
494 }
