725 template<typename ParameterType>
726 inline void ofParameter<ParameterType>::eventsSetValue(const ParameterType & v){
727 
728  // If the object is notifying its parents, just set the value without triggering an event.
729  if(obj->bInNotify)
730  {
731  noEventsSetValue(v);
732  }
733  else
734  {
735  // Mark the object as in its notification loop.
736  obj->bInNotify = true;
737 
738  // Set the value.
739  obj->value = v;
740 
741  // Notify any local subscribers.
742  ofNotifyEvent(obj->changedE,obj->value,this);
743 
744  // Notify all parents, if there are any.
745  if(!obj->parents.empty())
746  {
747  // Erase each invalid parent
748  obj->parents.erase(std::remove_if(obj->parents.begin(),
749  obj->parents.end(),
750  [this](const std::weak_ptr<ofParameterGroup::Value> & p){ return p.expired(); }),
751  obj->parents.end());
752 
753  // notify all leftover (valid) parents of this object's changed value.
754  // this can't happen in the same iterator as above, because a notified listener
755  // might perform similar cleanups that would corrupt our iterator
756  // (which appens for example if the listener calls getFirstParent on us)
757  for(auto & parent: obj->parents){
758  auto p = parent.lock();
759  if(p){
760  p->notifyParameterChanged(*this);
761  }
762  }
763  }
764  obj->bInNotify = false;
765  }
766 }
