18 // ofQTKitDecodeMode allows for highly efficient decoding of video
19 // pixel data. Options are available for direct-to-screen/texture rendering
20 // as well as more traditional CPU/RAM pixel-based rendering.
21 
22 enum ofQTKitDecodeMode {
23  OF_QTKIT_DECODE_PIXELS_ONLY = 0,
24  // OF_QTKIT_DECODE_PIXELS_ONLY: This mode decodes pixels only and can't
25  // be drawn directly to the screen without an an additional step.
26 
27  OF_QTKIT_DECODE_TEXTURE_ONLY,
28  // OF_QTKIT_DECODE_TEXTURE_ONLY: This is the fastest mode, but doesn't
29  // allow direct pixel access because video rendering skips the CPU and RAM.
30  // Pixels ARE available to texture / fragment shaders though.
31 
32  OF_QTKIT_DECODE_PIXELS_AND_TEXTURE
33  // OF_QTKIT_DECODE_PIXELS_AND_TEXTURE: This mode combines both
34  // OF_QTKIT_DECODE_PIXELS_ONLY AND OF_QTKIT_DECODE_TEXTURE_ONLY modes.
35  // It is slightly slower OF_QTKIT_DECODE_TEXTURE_ONLY, but faster than
36  // managing your own ofTexture.
37 };
